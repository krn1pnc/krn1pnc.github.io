#include <cstdio>

using i64 = long long;

inline i64 rd() {
	i64 x = 0, f = 1, c = getchar();
	while (((c - '0') | ('9' - c)) < 0)
		f = c != '-', c = getchar();
	while (((c - '0') | ('9' - c)) > 0)
		x = x * 10 + c - '0', c = getchar();
	return f ? x : -x;
}

const int N = 1e3;

i64 n, x, P, m, a[N + 5];
i64 S[N + 5][N + 5];
void pre() {
	S[0][0] = 1;
	for(int i = 1; i <= m; i++) {
		for(int j = 1; j <= m; j++) {
			S[i][j] = (S[i - 1][j - 1] + j * S[i - 1][j] % P) % P;
		}
	}
}

i64 fpow(i64 b, i64 p) {
	i64 res = 1;
	for(; p; b = b * b % P, p >>= 1) {
		if(p & 1) res = res * b % P;
	}
	return res;
}

int main() {
	n = rd(), x = rd(), P = rd(), m = rd(); pre();
	for(int i = 0; i <= m; i++) a[i] = rd();
	i64 ans = 0;
	for(int j = 0; j <= m; j++) {
		i64 nj = 1; for(int i = n - j + 1; i <= n; i++) (nj *= i) %= P;
		i64 fi = 0; for(int i = j; i <= m; i++) (fi += S[i][j] * a[i] % P) %= P;
		(ans += nj * fi % P * fpow(x, j) % P * fpow(x + 1, n - j) % P) %= P;
	}
	printf("%lld\n", ans);
	return 0;
}
